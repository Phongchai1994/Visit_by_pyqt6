def find_first_available_channel(used_channels, max_channel=14):
    used = set(int(c) for c in used_channels if c is not None)
    for ch in range(1, max_channel + 1):
        if ch not in used:
            return ch
    return None

ถ้าต้องการ "เติมช่องที่หายไป" (เลือกช่อง gap เล็กสุด) — ฟังก์ชันเดียวกันจะคืนช่องเล็กสุดที่ขาด (1..max)

การทำงานแบบปลอดภัย (transactional) — ไอเดีย
- เริ่ม transaction
- SELECT channel FROM visits WHERE date_visit=%s AND time_visit=%s FOR UPDATE
  (ล็อกแถวที่เกี่ยวข้องก่อนอ่าน)
- คำนวณช่องว่าง → เลือกช่องที่ต้องการ
- INSERT ใหม่ (หรือ UPDATE row เดิม) กับช่องที่เลือก
- Commit

@log_db_exceptions
def reserve_first_available_channel(self, date_visit, time_visit, prisoner_id, relative_ids, max_channel=14):
    """
    หาช่องว่างตัวแรกและบันทึกเป็น visit ใหม่ (transactional).
    `relative_ids` เป็น list ที่อาจมี 1..5 ids.
    คืนค่า channel ถ้าสำเร็จ หรือ None ถ้าเต็ม
    """
    with self.conn.cursor() as cur:
        # lock relevant rows to avoid race
        cur.execute(
            """
            SELECT channel FROM visits
            WHERE date_visit = %s AND time_visit = %s
            FOR UPDATE
            """,
            (date_visit, time_visit)
        )
        rows = cur.fetchall()
        used = {r[0] for r in rows if r[0] is not None}
        # find first free
        channel = None
        for ch in range(1, max_channel + 1):
            if ch not in used:
                channel = ch
                break
        if channel is None:
            return None
        # prepare relative columns (up to 5)
        rel_cols = ['relative_id_1','relative_id_2','relative_id_3','relative_id_4','relative_id_5']
        vals = [None]*5
        for i, rid in enumerate(relative_ids[:5]):
            vals[i] = rid
        cur.execute(
            f"""
            INSERT INTO visits (date_visit, time_visit, prisoner_id, {', '.join(rel_cols)}, channel)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            """,
            (date_visit, time_visit, prisoner_id, vals[0], vals[1], vals[2], vals[3], vals[4], channel)
        )
        return channel

ข้อควรระวังและปรับปรุงเพิ่มเติม
- ถ้าแอพมี concurrent สูง ให้ออกแบบ locking ให้ชัด (row-level locks หรือ advisory locks) และพิจารณา unique partial index เพื่อป้องกัน double-book:
  CREATE UNIQUE INDEX uniq_visit_channel ON visits (date_visit, time_visit, channel);
  แล้วใช้ INSERT และจับ UniqueViolation เพื่อ retry หาช่องถัดไป
- ถ้ต้องการ "เติมช่องให้ชิด" หลังการย้าย: เมื่อย้ายบันทึก ให้ UPDATE ช่องเก่าเป็น NULL แล้วระบบใหม่เรียกหา gap จะพบช่องว่างและเติมได้
- ถ้าต้องการ prioritization (เช่น อยากใช้ช่องต่ำสุด/สูงสุด หรือเฉพาะช่องบางกลุ่ม) ปรับลูปค้นหา accordingly
- ตรวจความสมบูรณ์ของ `relative_ids` ก่อน INSERT (max 5)
